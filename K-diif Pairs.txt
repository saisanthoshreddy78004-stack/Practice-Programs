def findTwoRepeating(arr, n):
    seen = set()
    repeats = []
    
    for num in arr:
        if num in seen:
            repeats.append(num)
            if len(repeats) == 2:
                break
        else:
            seen.add(num)
    
    return repeats


# Dynamic input
n = int(input("Enter N: "))
arr = list(map(int, input("Enter array elements separated by space: ").split()))

result = findTwoRepeating(arr, n)
print(*result)
